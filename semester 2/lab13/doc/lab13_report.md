``` 
int FindIndexElement(Type &element){
```

```
void deleteElement(int index){  
  if (index < 0) {  
  index = 0;  
    }  
  if (index >= this->index) {  
  index = this->index - 1;  
    }  
  for (int i = index; i < this->index; ++i) {  
  elements[i] = elements[i+1];  
    }  
  this->index--;  
}
```
